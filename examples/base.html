安装与使用

一、安装方法

npm install VirtualScroll --save

二、使用方法

页面引入
<script src="VirtualScroll.js"></script>
VirtualScroll(DOM对象);
RequireJS
require(['VirtualScroll'], function (VS){ VS(DOM对象); });
CommonJS
let VS = require('VirtualScroll.js');
VS(DOM对象);
ES2015
import VS from 'VirtualScroll';
VS(DOM对象);
